      - "--providers.docker.exposedbydefault=false"

    ports:

  nextcloud:
    image: nextcloud:29
    container_name: nextcloud-app
    restart: always
    depends_on:
      - nextcloud-db
      - nextcloud-redis
    environment:
      - POSTGRES_HOST=nextcloud-db
      - POSTGRES_PASSWORD=DBSxxxxxxx@
      - POSTGRES_DB=nextcloud

      - REDIS_HOST=nextcloud-redis
      - REDIS_HOST_PASSWORD=Rexxxxxxxx@

      - PHP_UPLOAD_LIMIT=1024M

      # ⚠ Se já tem dados em produção, planeje migração antes de alterar
      - nextcloud_main:/var/www/html
      - nextcloud_apps:/var/www/html/custom_apps
      - nextcloud_config:/var/www/html/config

      - root_backend
    labels:

      traefik.docker.network: "root_backend"

      traefik.http.routers.nextcloud.rule: "Host(`cloud.safion.com.br`)"
      traefik.http.routers.nextcloud.entrypoints: "websecure"
      traefik.http.routers.nextcloud.tls: "true"
      traefik.http.routers.nextcloud.tls.certresolver: "mytlschallenge"
      traefik.http.services.nextcloud.loadbalancer.server.port: "80"

      # HSTS + cabeçalhos de segurança
      traefik.http.middlewares.nextcloud-headers.headers.forceSTSHeader: "true"
      traefik.http.middlewares.nextcloud-headers.headers.STSSeconds: "15xxxxxx00"
      traefik.http.middlewares.nextcloud-headers.headers.STSIncludeSubdomains: "true"
      traefik.http.middlewares.nextcloud-headers.headers.STSPreload: "true"
      traefik.http.middlewares.nextcloud-headers.headers.browserXSSFilter: "true"
      traefik.http.middlewares.nextcloud-headers.headers.contentTypeNosniff: "true"
      traefik.http.middlewares.nextcloud-headers.headers.referrerPolicy: "no-referrer"

      # Redireciono CalDAV/CardDAV -> /remote.php/dav
      traefik.http.middlewares.nextcloud-dav.redirectregex.regex: "^/.well-known/(card|cal)dav$"
      traefik.http.middlewares.nextcloud-dav.redirectregex.replacement: "/remote.php/dav"
      traefik.http.middlewares.nextcloud-dav.redirectregex.permanent: "true"

      # Garante cabeçalhos corretos do proxy
      traefik.http.middlewares.sslheader.headers.customRequestHeaders.X-Forwarded-Proto: "https"
      traefik.http.middlewares.sslheader.headers.customRequestHeaders.X-Forwarded-For: "{client_ip}"
      traefik.http.middlewares.sslheader.headers.customRequestHeaders.X-Forwarded-Host: "cloud.safion.com.br"

      # Compressão (melhora bastante a performance)
      traefik.http.middlewares.nextcloud-compress.compress: "true"

      # Redireciono Webfinger -> obrigatório
      traefik.http.middlewares.nextcloud-webfinger.redirectregex.regex: "^/.well-known/webfinger$"
      traefik.http.middlewares.nextcloud-webfinger.redirectregex.replacement: "/index.php/.well-known/webfinger"
      traefik.http.middlewares.nextcloud-webfinger.redirectregex.permanent: "true"

      # Redireciono Nodeinfo -> obrigatório
      traefik.http.middlewares.nextcloud-nodeinfo.redirectregex.regex: "^/.well-known/nodeinfo$"
      traefik.http.middlewares.nextcloud-nodeinfo.redirectregex.replacement: "/index.php/.well-known/nodeinfo"
      traefik.http.middlewares.nextcloud-nodeinfo.redirectregex.permanent: "true"

      # Redireciono OCM Provider -> obrigatório
      traefik.http.middlewares.nextcloud-ocm.redirectregex.regex: "^/ocm-provider/?$"
      traefik.http.middlewares.nextcloud-ocm.redirectregex.replacement: "/ocs-provider/"
      traefik.http.middlewares.nextcloud-ocm.redirectregex.permanent: "true"

      # Adiciono todos os middlewares ao router Nextcloud
      traefik.http.routers.nextcloud.middlewares: "sslheader,nextcloud-headers,nextcloud-dav,nextcloud-compress,nextcloud-webfinger,nextcloud-nodeinfo,nextcloud-ocm"
